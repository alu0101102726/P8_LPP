def erase_element_head 
        if(@head != nil && @tail != nil) then
                if (@head == @tail) then
                        @head = nil
                        @tail = nil
                else
                        @head = @head[:next]
                        @head[:prev] = nil
                end

                return @head[:value]
        end
end

def expc_gei(n_dias)
        aux = @head
        suma = 0
        while aux != @tail[:next]
                suma += (aux[:value][0].gei * aux[:value][1])
                aux = aux[:next]
        end
        return (n_dias * suma).round(1)
end

    # File lib/p6/lista.rb, line 48
def insert_element_head( valor )
        if(@head == nil && @tail == nil) then
                node = Nodo.new(valor,nil,nil)
                @head = node
                @tail = node
        else
                node = Nodo.new(valor,@head,nil)
                @head[:prev] = node
                @head = node
        end

        return [node[:value], node[:next] , node[:prev]]
end
