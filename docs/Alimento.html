def imp_amb ( arry_gei , arry_terreno , cantidad)
        suma = 0
        suma = suma +(arry_gei[0] + arry_terreno[0])*cantidad[0]
        suma = suma +(arry_gei[1] + arry_terreno[1])*cantidad[1]
        suma = suma +(arry_gei[2] + arry_terreno[2])*cantidad[2]
        suma.round(1)
end
